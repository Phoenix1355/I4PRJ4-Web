<template lang="html">
    <div
        class="modal"
        :class="{ 'is-open': active }"
    >
        <div
            class="modal-overlay"
            @click="close"
        />
        <div class="modal-body">
            <div class="model-header">
                <button
                    class="modal-close"
                    @click="close"
                >
                    <span />
                    <span />
                </button>
            </div>
            <div class="modal-content">
                <slot name="content" />
            </div>
            <div class="modal-footer">
                <slot name="footer" />
            </div>
        </div>
    </div>
</template>

<script>
/**
 * A functional popup modal used for displaying content
 * @type {Component}
 */
export default {
    data: () => ({
        active: false,
    }),

    methods: {
        open() {
            console.log('Open() was called!');
            this.active = true;
        },
        close() {
            console.log('Close() was called!');
            this.active = false;
        },
    },
};
</script>
